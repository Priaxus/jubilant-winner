```sql
CREATE OR REPLACE FUNCTION audit_trigger() RETURNS trigger AS $$
BEGIN
    INSERT INTO audit_log VALUES (CURRENT_USER, CURRENT_TIMESTAMP, tg_op, NEW, OLD);
    RETURN NEW;
END;
$$
LANGUAGE PLPGSQL;

CREATE TRIGGER audit_trigger AFTER INSERT OR UPDATE OR DELETE ON t1 FOR EACH ROW EXECUTE FUNCTION audit_trigger();
```
